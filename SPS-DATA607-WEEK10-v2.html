For our Sentiment analysis we will use the Gutenberg Library and examine some of Henry Wadsworth Longfellow’s work.

hwl_books <- c("5436","23332","25153")


hwl_books_download <- gutenberg_download(hwl_books)
## Determining mirror for Project Gutenberg from http://www.gutenberg.org/robot/harvest
## Using mirror http://aleph.gutenberg.org
tidy_books <- hwl_books_download %>%
  group_by(hwl_books_download$gutenberg_id) %>%
  mutate(
    linenumber = row_number(),
    chapter = cumsum(str_detect(text, 
                                regex("^chapter [\\divxlc]", 
                                      ignore_case = TRUE)))) %>%
  ungroup() %>%
  unnest_tokens(word, text)


data("stop_words")

tidy_books <- tidy_books %>%
  anti_join(stop_words)
## Joining, by = "word"
